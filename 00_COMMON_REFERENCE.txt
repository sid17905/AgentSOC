# backend/schemas/incident.py
class Severity(str, Enum):
    P1 = "P1"   # Critical
    P2 = "P2"   # High
    P3 = "P3"   # Medium
    P4 = "P4"   # Low

class IncidentType(str, Enum):
    RANSOMWARE        = "ransomware"
    PHISHING          = "phishing"
    DATA_EXFILTRATION = "data_exfiltration"
    LATERAL_MOVEMENT  = "lateral_movement"
    BRUTE_FORCE       = "brute_force"
    UNKNOWN           = "unknown"

class IncidentStatus(str, Enum):
    PENDING           = "pending"
    ANALYZING         = "analyzing"
    AWAITING_APPROVAL = "awaiting_approval"
    APPROVED          = "approved"
    REJECTED          = "rejected"
    CLOSED            = "closed"

class IOC(BaseModel):
    type: str        # "ip" | "hash" | "domain" | "cve" | "url"
    value: str
    confidence: float
    reputation: Optional[str] = None   # from AbuseIPDB or offline list

class MITRETechnique(BaseModel):
    technique_id: str
    technique_name: str
    tactic: str

class IncidentReport(BaseModel):
    id: str
    status: IncidentStatus
    incident_type: IncidentType
    severity: Severity
    confidence: float
    title: str
    summary: str
    iocs: List[IOC]
    mitre_techniques: List[MITRETechnique]
    playbook_steps: List[str]
    raw_input: str
    created_at: datetime
    updated_at: datetime
    report_markdown: Optional[str] = None
    agent_thoughts: List[str] = []

# backend/schemas/agent_input.py
class InputType(str, Enum):
    LOG   = "log"
    PDF   = "pdf"
    IMAGE = "image"
    EMAIL = "email"
    JSON  = "json"

class AgentInput(BaseModel):
    input_type: InputType
    content: str
    filename: Optional[str] = None
